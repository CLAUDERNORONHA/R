                  x<- scan(n=1)
                  i<-1

                  while(x==1){
                  print("Digite o nome do aluno")
                  Nome<- scan(what = character(), nmax = 1)

                  Prova1<- scan(n=1)

                  print("Digite o resultado da segunda prova")
                  Prova2<- scan(n=1)

                  print("Digite o resultado da terceira prova")
                  Prova3<- scan(n=1)

                  Provas<- c(Prova1, Prova2, Prova3)
                  media<- mean(Provas)

                  if(media>=7){
                  situacao<- "Aprovado" }

                  if(media<7){
                  situacao<- "Reprovado" }

                  cat("O aluno ", Nome, "teve as seguintes notas: \n")
                  for(j in 1:length(Provas)){
                  cat("Prova ", j, "---> ", Provas[j], " \n")

                  cat("A média foi: ", media, "\n Sendo assim o aluno foi ", situacao, "\n\n")

                  Dados[i,]<- c(Nome, Prova1, Prova2, Prova3, media, situacao)
